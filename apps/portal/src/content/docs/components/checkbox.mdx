The `Checkbox` component provides a customizable, accessible checkbox input. It is built on top of the Radix UI Checkbox primitive with additional styling.

import { DocsPage } from "@/components/docs-page";

<DocsPage.ComponentExample
  client:only
  code={`() => {
    const [checked, setChecked] = React.useState(false);
    return (<div className="flex flex-col gap-4">
      <section class="flex items-center gap-3">
        <Checkbox id="checkboxId" checked={checked} onCheckedChange={setChecked} />
        <label className="Label" htmlFor="checkboxId">
          Accept terms and conditions.
        </label>
      </section>
    
      <section class="flex items-center gap-3">
        <Checkbox id="checkboxId2" checked={false} onCheckedChange={Function} disabled />
        <label className="Label" htmlFor="checkboxId2">
          Disabled checkbox
        </label>
      </section>
    </div>)
  }`}
/>

## Usage

```typescript jsx
import { Checkbox } from '@harnessio/ui/components'

return (
  <Checkbox id="checkbox" checked={true} onCheckedChange={(checked) => alert(\`Checked: \${checked}\`)} />
)
```
